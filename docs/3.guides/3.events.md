# Events and Observability

sctx emits [capitan](https://github.com/zoobzio/capitan) events for all significant operations. Use these for logging, metrics, alerting, and debugging.

## Event Categories

### Token Events

| Signal | Description | Fields |
|--------|-------------|--------|
| `TokenGenerated` | Token created from certificate | `fingerprint`, `common_name`, `permissions` |
| `TokenVerified` | Token successfully verified | `fingerprint` |
| `TokenRejected` | Token verification failed | `fingerprint`, `error` |

### Guard Events

| Signal | Description | Fields |
|--------|-------------|--------|
| `GuardCreated` | Permission guard created | `guard_id`, `fingerprint`, `required_permissions` |
| `GuardValidated` | Guard validation succeeded | `guard_id`, `fingerprint` |
| `GuardRejected` | Guard validation failed | `guard_id`, `fingerprint`, `error` |

### Context Events

| Signal | Description | Fields |
|--------|-------------|--------|
| `ContextRevoked` | Context manually revoked | `fingerprint` |

### Assertion Events

| Signal | Description | Fields |
|--------|-------------|--------|
| `AssertionValidated` | Assertion passed validation | `fingerprint` |
| `AssertionRejected` | Assertion validation failed | `fingerprint`, `error` |

### Certificate Events

| Signal | Description | Fields |
|--------|-------------|--------|
| `CertificateRejected` | Certificate verification failed | `common_name`, `error` |

### Cache Events

| Signal | Description | Fields |
|--------|-------------|--------|
| `CacheStored` | Context stored in cache | `fingerprint` |
| `CacheHit` | Cache lookup succeeded | `fingerprint` |
| `CacheMiss` | Cache lookup failed | `fingerprint` |
| `CacheDeleted` | Context deleted from cache | `fingerprint` |
| `CacheExpired` | Context expired during cleanup | `fingerprint` |

## Field Keys

All field keys are exported for use in listeners:

```go
sctx.FingerprintKey    // string - Certificate fingerprint
sctx.CommonNameKey     // string - Certificate common name
sctx.PermissionsKey    // string - Comma-separated permissions
sctx.RequiredPermsKey  // string - Required permissions for guard
sctx.GuardIDKey        // string - Guard unique identifier
sctx.ErrorKey          // string - Error message
sctx.ExpiresAtKey      // time.Time - Expiration timestamp
sctx.DurationMsKey     // int64 - Operation duration in milliseconds
```

## Basic Logging

Log all sctx events:

```go
import (
    "github.com/zoobzio/capitan"
    "github.com/zoobzio/sctx"
)

func setupLogging() {
    // Observe all sctx signals
    capitan.Observe(func(ctx context.Context, e *capitan.Event) {
        fields := make(map[string]any)

        if fp, ok := sctx.FingerprintKey.From(e); ok {
            fields["fingerprint"] = fp
        }
        if cn, ok := sctx.CommonNameKey.From(e); ok {
            fields["common_name"] = cn
        }
        if err, ok := sctx.ErrorKey.From(e); ok {
            fields["error"] = err
        }
        if guardID, ok := sctx.GuardIDKey.From(e); ok {
            fields["guard_id"] = guardID
        }

        log.Printf("[%s] %s %v", e.Severity(), e.Signal().Name(), fields)
    },
        sctx.TokenGenerated,
        sctx.TokenVerified,
        sctx.TokenRejected,
        sctx.GuardCreated,
        sctx.GuardValidated,
        sctx.GuardRejected,
        sctx.ContextRevoked,
    )
}
```

## Structured Logging with slog

```go
func setupSlogLogging() {
    logger := slog.Default()

    capitan.Observe(func(ctx context.Context, e *capitan.Event) {
        attrs := []slog.Attr{
            slog.String("signal", e.Signal().Name()),
            slog.Time("timestamp", e.Timestamp()),
        }

        if fp, ok := sctx.FingerprintKey.From(e); ok {
            attrs = append(attrs, slog.String("fingerprint", fp[:16]+"..."))
        }
        if cn, ok := sctx.CommonNameKey.From(e); ok {
            attrs = append(attrs, slog.String("common_name", cn))
        }
        if err, ok := sctx.ErrorKey.From(e); ok {
            attrs = append(attrs, slog.String("error", err))
        }

        switch e.Severity() {
        case capitan.SeverityError:
            logger.LogAttrs(ctx, slog.LevelError, e.Signal().Description(), attrs...)
        case capitan.SeverityWarn:
            logger.LogAttrs(ctx, slog.LevelWarn, e.Signal().Description(), attrs...)
        default:
            logger.LogAttrs(ctx, slog.LevelInfo, e.Signal().Description(), attrs...)
        }
    })
}
```

## Metrics Collection

Export Prometheus metrics:

```go
var (
    tokensGenerated = prometheus.NewCounterVec(
        prometheus.CounterOpts{
            Name: "sctx_tokens_generated_total",
            Help: "Total tokens generated",
        },
        []string{"common_name"},
    )

    tokenVerifications = prometheus.NewCounterVec(
        prometheus.CounterOpts{
            Name: "sctx_token_verifications_total",
            Help: "Total token verifications",
        },
        []string{"status"},
    )

    guardValidations = prometheus.NewCounterVec(
        prometheus.CounterOpts{
            Name: "sctx_guard_validations_total",
            Help: "Total guard validations",
        },
        []string{"status"},
    )

    cacheOperations = prometheus.NewCounterVec(
        prometheus.CounterOpts{
            Name: "sctx_cache_operations_total",
            Help: "Cache operations",
        },
        []string{"operation"},
    )
)

func setupMetrics() {
    prometheus.MustRegister(tokensGenerated, tokenVerifications, guardValidations, cacheOperations)

    capitan.Hook(sctx.TokenGenerated, func(ctx context.Context, e *capitan.Event) {
        cn, _ := sctx.CommonNameKey.From(e)
        tokensGenerated.WithLabelValues(cn).Inc()
    })

    capitan.Hook(sctx.TokenVerified, func(ctx context.Context, e *capitan.Event) {
        tokenVerifications.WithLabelValues("success").Inc()
    })

    capitan.Hook(sctx.TokenRejected, func(ctx context.Context, e *capitan.Event) {
        tokenVerifications.WithLabelValues("failure").Inc()
    })

    capitan.Hook(sctx.GuardValidated, func(ctx context.Context, e *capitan.Event) {
        guardValidations.WithLabelValues("success").Inc()
    })

    capitan.Hook(sctx.GuardRejected, func(ctx context.Context, e *capitan.Event) {
        guardValidations.WithLabelValues("failure").Inc()
    })

    capitan.Hook(sctx.CacheHit, func(ctx context.Context, e *capitan.Event) {
        cacheOperations.WithLabelValues("hit").Inc()
    })

    capitan.Hook(sctx.CacheMiss, func(ctx context.Context, e *capitan.Event) {
        cacheOperations.WithLabelValues("miss").Inc()
    })
}
```

## Security Alerting

Alert on suspicious activity:

```go
func setupSecurityAlerting(alerter Alerter) {
    // Alert on repeated token rejections
    rejectionCounter := make(map[string]int)
    var mu sync.Mutex

    capitan.Hook(sctx.TokenRejected, func(ctx context.Context, e *capitan.Event) {
        fp, ok := sctx.FingerprintKey.From(e)
        if !ok {
            return
        }

        mu.Lock()
        rejectionCounter[fp]++
        count := rejectionCounter[fp]
        mu.Unlock()

        if count >= 5 {
            errMsg, _ := sctx.ErrorKey.From(e)
            alerter.Alert("Repeated token rejection", map[string]any{
                "fingerprint": fp,
                "count":       count,
                "error":       errMsg,
            })
        }
    })

    // Alert on certificate rejections
    capitan.Hook(sctx.CertificateRejected, func(ctx context.Context, e *capitan.Event) {
        cn, _ := sctx.CommonNameKey.From(e)
        errMsg, _ := sctx.ErrorKey.From(e)
        alerter.Alert("Certificate rejected", map[string]any{
            "common_name": cn,
            "error":       errMsg,
        })
    })

    // Alert on context revocations
    capitan.Hook(sctx.ContextRevoked, func(ctx context.Context, e *capitan.Event) {
        fp, _ := sctx.FingerprintKey.From(e)
        alerter.Alert("Context revoked", map[string]any{
            "fingerprint": fp,
        })
    })
}
```

## Audit Trail

Create an audit log:

```go
type AuditEntry struct {
    Timestamp   time.Time
    Event       string
    Fingerprint string
    CommonName  string
    Action      string
    Success     bool
    Error       string
}

func setupAuditLog(store AuditStore) {
    auditSignals := []capitan.Signal{
        sctx.TokenGenerated,
        sctx.TokenRejected,
        sctx.GuardCreated,
        sctx.GuardRejected,
        sctx.ContextRevoked,
    }

    capitan.Observe(func(ctx context.Context, e *capitan.Event) {
        entry := AuditEntry{
            Timestamp: e.Timestamp(),
            Event:     e.Signal().Name(),
        }

        if fp, ok := sctx.FingerprintKey.From(e); ok {
            entry.Fingerprint = fp
        }
        if cn, ok := sctx.CommonNameKey.From(e); ok {
            entry.CommonName = cn
        }
        if err, ok := sctx.ErrorKey.From(e); ok {
            entry.Error = err
            entry.Success = false
        } else {
            entry.Success = true
        }

        store.Store(ctx, entry)
    }, auditSignals...)
}
```

## Debugging

Enable verbose logging for debugging:

```go
func enableDebugLogging() {
    // Log all events including cache operations
    capitan.Observe(func(ctx context.Context, e *capitan.Event) {
        var fields []string
        for _, f := range e.Fields() {
            fields = append(fields, fmt.Sprintf("%s=%v", f.Key().Name(), f.Value()))
        }
        log.Printf("DEBUG [%s] %s: %s", e.Signal().Name(), e.Signal().Description(), strings.Join(fields, " "))
    })
}
```

## Testing with Events

Capture events in tests:

```go
func TestTokenGeneration(t *testing.T) {
    var generated bool
    var capturedFingerprint string

    listener := capitan.Hook(sctx.TokenGenerated, func(ctx context.Context, e *capitan.Event) {
        generated = true
        capturedFingerprint, _ = sctx.FingerprintKey.From(e)
    })
    defer listener.Close()

    // ... generate token ...

    // Drain events
    capitan.Shutdown()

    assert.True(t, generated)
    assert.NotEmpty(t, capturedFingerprint)
}
```
