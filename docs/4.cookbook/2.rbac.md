# Role-Based Access Control

*Recipe: Implement RBAC using certificate attributes and context policies.*

## The Scenario

You need role-based access control where:
- Roles are determined by certificate attributes
- Each role has a set of permissions
- Permissions grant access to resources and actions

## Role Definition

Define roles and their permissions:

```go
package rbac

// Role represents a user role
type Role string

const (
    RoleAdmin     Role = "admin"
    RoleManager   Role = "manager"
    RoleDeveloper Role = "developer"
    RoleViewer    Role = "viewer"
)

// RolePermissions maps roles to permissions
var RolePermissions = map[Role][]string{
    RoleAdmin: {
        "users:create", "users:read", "users:update", "users:delete",
        "projects:create", "projects:read", "projects:update", "projects:delete",
        "settings:read", "settings:write",
        "audit:read",
    },
    RoleManager: {
        "users:read", "users:update",
        "projects:create", "projects:read", "projects:update",
        "reports:read",
    },
    RoleDeveloper: {
        "projects:read", "projects:update",
        "code:read", "code:write",
        "deployments:read", "deployments:create",
    },
    RoleViewer: {
        "projects:read",
        "reports:read",
    },
}

// GetPermissions returns permissions for a role
func GetPermissions(role Role) []string {
    return RolePermissions[role]
}
```

## Role Extraction from Certificate

Extract role from certificate attributes:

```go
package rbac

import (
    "crypto/x509"
    "strings"
)

// ExtractRole determines role from certificate
func ExtractRole(cert *x509.Certificate) Role {
    // Check organizational units for role indicators
    for _, ou := range cert.Subject.OrganizationalUnit {
        ou = strings.ToLower(ou)
        switch {
        case strings.Contains(ou, "admin"):
            return RoleAdmin
        case strings.Contains(ou, "manager"):
            return RoleManager
        case strings.Contains(ou, "developer"), strings.Contains(ou, "engineering"):
            return RoleDeveloper
        }
    }

    // Check for role in email domain
    for _, email := range cert.EmailAddresses {
        if strings.HasSuffix(email, "@admin.company.com") {
            return RoleAdmin
        }
    }

    // Check certificate extensions for custom role field
    for _, ext := range cert.Extensions {
        if ext.Id.Equal(roleExtensionOID) {
            return Role(string(ext.Value))
        }
    }

    // Default to viewer
    return RoleViewer
}
```

## Context Policy

Create a policy that maps roles to permissions:

```go
package main

import (
    "crypto/x509"
    "time"

    "github.com/zoobzio/sctx"
    "myapp/rbac"
)

type UserMeta struct {
    Role     rbac.Role
    Username string
    Groups   []string
}

func rbacPolicy(cert *x509.Certificate) (*sctx.Context[UserMeta], error) {
    // Extract role from certificate
    role := rbac.ExtractRole(cert)

    // Get permissions for role
    permissions := rbac.GetPermissions(role)

    // Extract groups from OUs
    var groups []string
    for _, ou := range cert.Subject.OrganizationalUnit {
        if !isRoleOU(ou) {
            groups = append(groups, ou)
        }
    }

    return &sctx.Context[UserMeta]{
        Permissions: permissions,
        ExpiresAt:   time.Now().Add(time.Hour),
        Metadata: UserMeta{
            Role:     role,
            Username: cert.Subject.CommonName,
            Groups:   groups,
        },
    }, nil
}
```

## Permission Checking

Create guards for different permission requirements:

```go
func setupRoutes(admin sctx.Admin[UserMeta]) {
    // User management - admin only
    http.Handle("/api/users", requirePermissions(admin, "users:read"))
    http.Handle("/api/users/create", requirePermissions(admin, "users:create"))

    // Projects - multiple roles
    http.Handle("/api/projects", requirePermissions(admin, "projects:read"))
    http.Handle("/api/projects/create", requirePermissions(admin, "projects:create"))

    // Code - developers
    http.Handle("/api/code", requirePermissions(admin, "code:read"))
    http.Handle("/api/code/commit", requirePermissions(admin, "code:write"))

    // Settings - admin only
    http.Handle("/api/settings", requirePermissions(admin, "settings:read"))
}

func requirePermissions(admin sctx.Admin[UserMeta], perms ...string) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        token := extractToken(r)

        guard, err := admin.CreateGuard(r.Context(), token, perms...)
        if err != nil {
            http.Error(w, "Forbidden", http.StatusForbidden)
            return
        }

        if err := guard.Validate(r.Context(), token); err != nil {
            http.Error(w, "Forbidden", http.StatusForbidden)
            return
        }

        // Continue to handler...
    })
}
```

## Role Hierarchy

Implement role inheritance:

```go
// RoleHierarchy defines which roles inherit from others
var RoleHierarchy = map[Role][]Role{
    RoleAdmin:     {RoleManager, RoleDeveloper, RoleViewer},
    RoleManager:   {RoleViewer},
    RoleDeveloper: {RoleViewer},
    RoleViewer:    {},
}

// GetEffectivePermissions returns all permissions including inherited
func GetEffectivePermissions(role Role) []string {
    seen := make(map[string]bool)
    var perms []string

    var collect func(r Role)
    collect = func(r Role) {
        for _, p := range RolePermissions[r] {
            if !seen[p] {
                seen[p] = true
                perms = append(perms, p)
            }
        }
        for _, inherited := range RoleHierarchy[r] {
            collect(inherited)
        }
    }

    collect(role)
    return perms
}
```

Update the policy to use effective permissions:

```go
func rbacPolicyWithHierarchy(cert *x509.Certificate) (*sctx.Context[UserMeta], error) {
    role := rbac.ExtractRole(cert)
    permissions := rbac.GetEffectivePermissions(role)  // Includes inherited

    return &sctx.Context[UserMeta]{
        Permissions: permissions,
        ExpiresAt:   time.Now().Add(time.Hour),
        Metadata: UserMeta{
            Role:     role,
            Username: cert.Subject.CommonName,
        },
    }, nil
}
```

## Dynamic Role Assignment

Look up roles from external system:

```go
func dynamicRBACPolicy(roleService RoleService) sctx.ContextPolicy[UserMeta] {
    return func(cert *x509.Certificate) (*sctx.Context[UserMeta], error) {
        ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
        defer cancel()

        // Look up role from external service
        role, err := roleService.GetRole(ctx, cert.Subject.CommonName)
        if err != nil {
            return nil, fmt.Errorf("failed to get role: %w", err)
        }

        // Get permissions for role
        permissions, err := roleService.GetPermissions(ctx, role)
        if err != nil {
            return nil, fmt.Errorf("failed to get permissions: %w", err)
        }

        return &sctx.Context[UserMeta]{
            Permissions: permissions,
            ExpiresAt:   time.Now().Add(30 * time.Minute), // Shorter for dynamic
            Metadata: UserMeta{
                Role:     Role(role),
                Username: cert.Subject.CommonName,
            },
        }, nil
    }
}
```

## Role-Based UI

Include role in API responses for UI:

```go
func meHandler(admin sctx.Admin[UserMeta]) http.HandlerFunc {
    return func(w http.ResponseWriter, r *http.Request) {
        token := extractToken(r)

        // Get context from cache
        fingerprint := extractFingerprintFromToken(token)
        ctx, ok := admin.GetContext(r.Context(), fingerprint)
        if !ok {
            http.Error(w, "Context not found", http.StatusUnauthorized)
            return
        }

        json.NewEncoder(w).Encode(map[string]any{
            "username":    ctx.Metadata.Username,
            "role":        ctx.Metadata.Role,
            "permissions": ctx.Permissions,
            "groups":      ctx.Metadata.Groups,
            "expires_at":  ctx.ExpiresAt,
        })
    }
}
```

## What This Produces

1. Certificates contain role indicators (OUs, extensions, or email domains)
2. Policy extracts role and maps to permissions
3. Guards check permissions, not roles
4. Role hierarchy provides permission inheritance
5. External services can provide dynamic role assignment
